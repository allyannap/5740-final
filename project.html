# check whether an ICD-9 diagnosis code indicates diabetes
is_diabetes_dx <- function(code) {
  code <- as.character(code)
  ifelse(
    is.na(code),
    FALSE,
    startsWith(code,"250")
  )
}

# recoding primary diagnosis (encoded) into 8 meaningful groups
diag_category <- function(code) {
  if (is.na(code)) {
    return("Other")
  }
  code <- as.character(code)
  if (startsWith(code, "V") || startsWith(code, "E")) {
    return("Other")
  }
  num <- suppressWarnings(as.numeric(code))
  if (is.na(num)) {
    return("Other")
  } else if ((num>=390 & num<=459) | num==785) {
    return("Circulatory")
  } else if ((num>=460 & num<=519) | num==786) {
    return("Respiratory")
  } else if ((num>=520 & num<=579) | num==787) {
    return("Digestive")
  } else if (floor(num)==250) {
    return("Diabetes")
  } else if (num>=800 & num<=999) {
    return("Injury")
  } else if (num>=710 & num<=739) {
    return("Musculoskeletal")
  } else if ((num>=580 & num<=629) | num==788) {
    return("Genitourinary")
  } else if (num>=140 & num<=239) {
    return("Neoplasms")
  } else {
    return("Other")
  }
}
